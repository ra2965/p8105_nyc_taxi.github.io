Heatmap of time (hour) by trip destination

11/30/2019

heatmap <-
  read_csv("./data/pilot.csv")
## Warning: Missing column names filled in: 'X1' [1]
## Parsed with column specification:

##   .default = col_double(),
##   store_and_fwd_flag = col_character(),

##   pu_zone = col_character(),
##   do_boro = col_character(),
##   do_zone = col_character()
## )
## See spec(...) for full column specifications.
heatmap %>% 
  filter(do_boro != "Unknown") %>% 
  select (do_boro, pu_hour) %>% 
  mutate(
    do_boro = fct_infreq(do_boro)
  ) %>% 
  group_by(do_boro, pu_hour) %>% 
  count() %>% 
  group_by(do_boro) %>% 
  mutate( 
    total = sum(n)
    ) %>% 
   mutate(percent = round((n/total)*100,1)) %>% 
  arrange(do_boro, pu_hour) %>% 
    plot_ly(
    x = ~pu_hour, y = ~do_boro,
    height = 400, width = 1400) %>%
  add_heatmap(
    z = ~percent,
    text = ~paste(
      "Trip to: ", do_boro, "<br>Time: ", pu_hour, ": 00",
      "<br>Percent: ", percent, "%"),
    hoverinfo = "text", showscale = TRUE,
    opacity = 0.90)
